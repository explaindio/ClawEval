```python
import time
from collections import OrderedDict
from typing import Any, Dict, List, Optional


class LRUCache:
    def __init__(self, capacity: int):
        if capacity <= 0:
            raise ValueError("capacity must be > 0")
        self._capacity: int = capacity
        self._cache: OrderedDict[str, tuple[Any, Optional[float]]] = OrderedDict()
        self._hits: int = 0
        self._misses: int = 0
        self._evictions: int = 0
        self._expirations: int = 0

    # ---------- internal helpers ----------
    def _now(self) -> float:
        return time.time()

    def _is_expired(self, expire_at: Optional[float]) -> bool:
        return expire_at is not None and expire_at <= self._now()

    def _remove_if_expired(self, key: str) -> bool:
        """Return True if the key was removed because it expired."""
        value, expire_at = self._cache.get(key, (None, None))
        if expire_at is not None and self._is_expired(expire_at):
            del self._cache[key]
            self._expirations += 1
            return True
        return False

    def _purge_expired(self) -> None:
        """Remove all expired entries (used lazily in size/keys)."""
        for key in list(self._cache.keys()):
            if self._remove_if_expired(key):
                continue

    # ---------- public API ----------
    def get(self, key: str) -> Any:
        if key not in self._cache:
            self._misses += 1
            return None
        if self._remove_if_expired(key):
            self._misses += 1
            return None
        # not expired
        value, _ = self._cache[key]
        self._cache.move_to_end(key)  # mark as most‑recently used
        self._hits += 1
        return value

    def put(self, key: str, value: Any, ttl: float = None) -> None:
        expire_at = None
        if ttl is not None:
            if ttl < 0:
                raise ValueError("ttl must be non‑negative")
            expire_at = self._now() + ttl

        # If key exists, just update
        if key in self._cache:
            self._cache[key] = (value, expire_at)
            self._cache.move_to_end(key)
            return

        # New entry: ensure capacity
        while len(self._cache) >= self._capacity:
            # evict LRU (first item)
            lru_key, (lru_val, lru_exp) = self._cache.popitem(last=False)
            if lru_exp is not None and self._is_expired(lru_exp):
                self._expirations += 1
            else:
                self._evictions += 1
        self._cache[key] = (value, expire_at)

    def delete(self, key: str) -> bool:
        if key not in self._cache:
            return False
        if self._remove_if_expired(key):
            return False
        del self._cache[key]
        return True

    def size(self) -> int:
        self._purge_expired()
        return len(self._cache)

    def get_stats(self) -> Dict[str, int]:
        return {
            "hits": self._hits,
            "misses": self._misses,
            "evictions": self._evictions,
            "expirations": self._expirations,
        }

    def keys(self) -> List[str]:
        self._purge_expired()
        # OrderedDict stores LRU at beginning, MRU at end
        return list(reversed(self._cache.keys()))

    def clear(self) -> None:
        self._cache.clear()
        self._hits = self._misses = self._evictions = self._expirations = 0
```
