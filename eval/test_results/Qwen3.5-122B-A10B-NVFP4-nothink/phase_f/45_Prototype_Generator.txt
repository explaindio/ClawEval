```python
from flask import Flask, request, jsonify
from dataclasses import dataclass
from typing import Optional, List, Dict, Any
from datetime import datetime

app = Flask(__name__)

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float
    created_at: str

books_db: List[Dict[str, Any]] = []
next_id = 1

def validate_book_data(data: Dict[str, Any], required_fields: List[str]) -> Optional[str]:
    missing_fields = [field for field in required_fields if field not in data or not data[field]]
    if missing_fields:
        return f"Missing required fields: {', '.join(missing_fields)}"
    
    if 'price' in data:
        try:
            price = float(data['price'])
            if price < 0:
                return "Price must be a non-negative number"
        except (ValueError, TypeError):
            return "Price must be a valid number"
    
    if 'isbn' in data and not isinstance(data['isbn'], str):
        return "ISBN must be a string"
        
    return None

@app.route('/books', methods=['GET'])
def get_books():
    page = request.args.get('page', 1, type=int)
    per_page = request.args.get('per_page', 10, type=int)
    
    if page < 1 or per_page < 1:
        return jsonify({"error": "page and per_page must be positive integers"}), 400
    
    start = (page - 1) * per_page
    end = start + per_page
    
    paginated_books = books_db[start:end]
    
    return jsonify({
        "books": paginated_books,
        "page": page,
        "per_page": per_page,
        "total": len(books_db)
    }), 200

@app.route('/books/<int:book_id>', methods=['GET'])
def get_book(book_id):
    book = next((b for b in books_db if b['id'] == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(book), 200

@app.route('/books', methods=['POST'])
def create_book():
    global next_id
    
    if not request.is_json:
        return jsonify({"error": "Content-Type must be application/json"}), 400
    
    data = request.get_json()
    required_fields = ['title', 'author', 'isbn', 'price']
    
    error = validate_book_data(data, required_fields)
    if error:
        return jsonify({"error": error}), 422
    
    new_book = {
        "id": next_id,
        "title": data['title'],
        "author": data['author'],
        "isbn": data['isbn'],
        "price": float(data['price']),
        "created_at": datetime.utcnow().isoformat()
    }
    
    books_db.append(new_book)
    next_id += 1
    
    return jsonify(new_book), 201

@app.route('/books/<int:book_id>', methods=['PUT'])
def update_book(book_id):
    if not request.is_json:
        return jsonify({"error": "Content-Type must be application/json"}), 400
    
    data = request.get_json()
    required_fields = ['title', 'author', 'isbn', 'price']
    
    error = validate_book_data(data, required_fields)
    if error:
        return jsonify({"error": error}), 422
    
    book_index = next((i for i, b in enumerate(books_db) if b['id'] == book_id), None)
    if book_index is None:
        return jsonify({"error": "Book not found"}), 404
    
    updated_book = {
        "id": book_id,
        "title": data['title'],
        "author": data['author'],
        "isbn": data['isbn'],
        "price": float(data['price']),
        "created_at": books_db[book_index]['created_at']
    }
    
    books_db[book_index] = updated_book
    return jsonify(updated_book), 200

@app.route('/books/<int:book_id>', methods=['DELETE'])
def delete_book(book_id):
    global books_db
    
    book_index = next((i for i, b in enumerate(books_db) if b['id'] == book_id), None)
    if book_index is None:
        return jsonify({"error": "Book not found"}), 404
    
    deleted_book = books_db.pop(book_index)
    return jsonify({"message": "Book deleted successfully", "book": deleted_book}), 200

if __name__ == '__main__':
    app.run(debug=True)
```
