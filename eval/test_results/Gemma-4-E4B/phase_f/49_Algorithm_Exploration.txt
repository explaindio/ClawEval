class Node:
    """Represents a node in the doubly linked list."""
    def __init__(self, key, value):
        self.key = key
        self.value = value
        self.prev = None
        self.next = None

class LRUCache:
    """
    Implements an LRU Cache using a dictionary for O(1) lookups 
    and a doubly linked list for O(1) order maintenance.
    """
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.cache = {}  # key -> Node
        
        # Dummy head and tail nodes for the doubly linked list
        self.head = Node(0, 0)
        self.tail = Node(0, 0)
        
        # Initialize the list: head <-> tail
        self.head.next = self.tail
        self.tail.prev = self.head

    def _remove_node(self, node: Node):
        """Removes a node from the linked list."""
        prev_node = node.prev
        next_node = node.next
        
        prev_node.next = next_node
        next_node.prev = prev_node

    def _add_to_head(self, node: Node):
        """Adds a node right after the head (making it MRU)."""
        # Node -> Head.next
        # Head -> Node
        
        node.prev = self.head
        node.next = self.head.next
        
        self.head.next.prev = node
        self.head.next = node

    def get(self, key: int) -> int:
        """
        Retrieves the value for a key. O(1).
        Returns -1 if not found.
        """
        if key not in self.cache:
            return -1
        
        node = self.cache[key]
        
        # Move the accessed node to the head (MRU)
        self._remove_node(node)
        self._add_to_head(node)
        
        return node.value

    def put(self, key: int, value: int) -> None:
        """
        Inserts or updates a key-value pair. O(1).
        Evicts LRU if capacity is reached.
        """
        if key in self.cache:
            # Update existing node
            node = self.cache[key]
            node.value = value
            
            # Move to MRU position
            self._remove_node(node)
            self._add_to_head(node)
            return

        # New node insertion
        new_node = Node(key, value)
        self.cache[key] = new_node
        self._add_to_head(new_node)

        # Check capacity
        if len(self.cache) > self.capacity:
            # Evict the Least Recently Used item (the node before the tail)
            lru_node = self.tail.prev
            
            # Remove from DLL
            self._remove_node(lru_node)
            
            # Remove from cache dictionary
            del self.cache[lru_node.key]


def lru_cache(capacity: int) -> LRUCache:
    """
    Factory function that returns an LRUCache object.
    """
    return LRUCache(capacity)
