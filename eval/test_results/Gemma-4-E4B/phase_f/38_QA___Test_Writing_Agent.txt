import pytest
from your_module import calculate_tax # Assuming the function is in 'your_module.py'

# --- Input Validation Tests (Edge Cases) ---

def test_invalid_income_negative():
    """Test case for negative income."""
    with pytest.raises(ValueError, match="Income must be non-negative number"):
        calculate_tax(-100, 'single')

def test_invalid_income_type():
    """Test case for non-numeric income input."""
    with pytest.raises(ValueError, match="Income must be non-negative number"):
        calculate_tax("abc", 'single')

def test_invalid_filing_status():
    """Test case for unsupported filing status."""
    with pytest.raises(ValueError, match="Invalid filing status"):
        calculate_tax(50000, 'divorced')

# --- Zero and Low Income Tests (Edge Cases) ---

def test_zero_income():
    """Test case where income is exactly zero."""
    tax, rate = calculate_tax(0, 'single')
    assert tax == 0.00
    assert rate == 0.00

def test_income_below_first_bracket():
    """Test case where income is very low (e.g., below the 10% bracket)."""
    # Single bracket starts at 11600 (10% rate)
    income = 5000
    tax, rate = calculate_tax(income, 'single')
    # Tax should be 0 since the first bracket starts higher
    assert tax == 0.00
    assert rate == 0.00

# --- Typical and Boundary Tests ---

def test_typical_income_single():
    """Test a typical income level for a single filer (mid-bracket)."""
    # Income: 50000
    # 1. 11600 @ 10% = 1160.00
    # 2. (47150 - 11600) = 35550 @ 12% = 4266.00
    # 3. (50000 - 47150) = 2850 @ 22% = 627.00
    # Total Tax = 1160 + 4266 + 627 = 6053.00
    # Effective Rate = 6053 / 50000 * 100 = 12.106 -> 12.11
    tax, rate = calculate_tax(50000, 'single')
    assert tax == 6053.00
    assert rate == 12.11

def test_bracket_boundary_single_12_percent():
    """Test income exactly at the boundary of the 10% to 12% bracket (47150)."""
    # Income: 47150
    # 1. 11600 @ 10% = 1160.00
    # 2. (47150 - 11600) = 35550 @ 12% = 4266.00
    # Total Tax = 1160 + 4266 = 5426.00
    # Effective Rate = 5426 / 47150 * 100 = 11.50
    tax, rate = calculate_tax(47150, 'single')
    assert tax == 5426.00
    assert rate == 11.50

# --- High Income and Comparison Tests ---

def test_high_income_single():
    """Test a very high income level (hitting the top 37% bracket)."""
    # Income: 1_000_000

    # Tax for $1M single is $335,065.00
    tax, rate = calculate_tax(1_000_000, 'single')
    assert tax == 335065.00
    # Effective Rate = 335065 / 1000000 * 100 = 33.51
    assert rate == 33.51

def test_high_income_married():

def test_single_vs_married_difference():
    """Test a mid-range income to ensure the difference between filing statuses is captured."""
    income = 150000
    
    # Single Tax (approx): $34,995.00
    tax_single, _ = calculate_tax(income, 'single')
    
    # Married Tax (approx): $28,585.00
    tax_married, _ = calculate_tax(income, 'married')
    
    # Married tax should be significantly lower than single tax for the same income
    assert tax_married < tax_single
