```python
from flask import Flask, request, jsonify
from dataclasses import dataclass
from typing import List, Optional

app = Flask(__name__)

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

# In-memory database
books: List[Book] = []
next_id = 1

def get_error_response(message: str, status_code: int = 400):
    """Helper function to return standardized JSON error responses."""
    return jsonify({"error": message}), status_code

# --- Endpoints ---

@app.route('/books', methods=['GET'])
def list_books():
    """
    GET /books - List all books with pagination.
    Query parameters: ?page=1&per_page=10
    """
    try:
        page = int(request.args.get('page', 1))
        per_page = int(request.args.get('per_page', 10))
    except ValueError:
        return get_error_response("Page and per_page must be integers.", 400)

    if page < 1 or per_page < 1:
        return get_error_response("Page and per_page must be positive.", 400)

    start = (page - 1) * per_page
    end = start + per_page

    paginated_books = books[start:end]
    total_books = len(books)

    return jsonify({
        "books": [book.__dict__ for book in paginated_books],
        "total_items": total_books,
        "page": page,
        "per_page": per_page
    }), 200

@app.route('/books/<int:book_id>', methods=['GET'])
def get_book(book_id: int):
    """
    GET /books/<id> - Get a single book.
    """
    book = next((b for b in books if b.id == book_id), None)
    if book is None:
        return get_error_response(f"Book with ID {book_id} not found.", 404)
    
    return jsonify(book.__dict__), 200

@app.route('/books', methods=['POST'])
def create_book():
    """
    POST /books - Create a new book.
    Requires: title, author, isbn, price
    """
    data = request.get_json()
    if not data:
        return get_error_response("Invalid JSON payload.", 400)

    required_fields = ['title', 'author', 'isbn', 'price']
    if not all(field in data for field in required_fields):
        return get_error_response("Missing required fields: title, author, isbn, price.", 422)

    try:
        price = float(data['price'])
        if price <= 0:
             return get_error_response("Price must be positive.", 422)
    except ValueError:
        return get_error_response("Price must be a valid number.", 422)

    new_book = Book(
        id=next_id,
        title=data['title'],
        author=data['author'],
        isbn=data['isbn'],
        price=price
    )
    
    books.append(new_book)
    global next_id
    next_id += 1
    
    return jsonify(new_book.__dict__), 201

@app.route('/books/<int:book_id>', methods=['PUT'])
def update_book(book_id: int):
    """
    PUT /books/<id> - Update an existing book.
    """
    book_index = next((i for i, book in enumerate(books) if book.id == book_id), -1)
    
    if book_index == -1:
        return get_error_response(f"Book with ID {book_id} not found.", 404)

    data = request.get_json()
    if not data:
        return get_error_response("Invalid JSON payload.", 400)

    try:
        # Update fields if provided
        if 'title' in data:
            books[book_index].title = data['title']
        if 'author' in data:
            books[book_index].author = data['author']
        if 'isbn' in data:
            books[book_index].isbn = data['isbn']
        if 'price' in data:
            price = float(data['price'])
            if price <= 0:
                return get_error_response("Price must be positive.", 422)
            books[book_index].price = price
        
        return jsonify(books[book_index].__dict__), 200

    except ValueError:
        return get_error_response("Invalid data type provided (e.g., price must be a number).", 422)


@app.route('/books/<int:book_id>', methods=['DELETE'])
def delete_book(book_id: int):
    """
    DELETE /books/<id> - Delete a book.
    """
    global books
    initial_length = len(books)
    
    # Filter out the book to be deleted
    books = [book for book in books if book.id != book_id]
    
    if len(books) == initial_length:
        return get_error_response(f"Book with ID {book_id} not found.", 404)
    
    # 204 No Content is standard for successful deletion
    return '', 204

if __name__ == '__main__':
    # Example initialization data (optional)
    books.append(Book(id=1, title="The Great Gatsby", author="F. Scott Fitzgerald", isbn="978-0743273565", price=10.99))
    books.append(Book(id=2, title="1984", author="George Orwell", isbn="978-0451524935", price=12.50))
    
    app.run(debug=True)
```
