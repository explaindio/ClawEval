from flask import Flask, request, jsonify
from dataclasses import dataclass, asdict
from typing import List, Optional
import re

app = Flask(__name__)

# In-memory database
books_db: List[dict] = []
next_id = 1

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

def validate_book_data(data: dict) -> Optional[dict]:
    """Validates book data and returns cleaned data or None if invalid."""
    required_fields = ['title', 'author', 'isbn', 'price']
    
    for field in required_fields:
        if field not in data:
            return None
    
    # Validate price is a number
    try:
        price = float(data['price'])
        if price < 0:
            return None
        data['price'] = price
    except (ValueError, TypeError):
        return None
    
    # Validate ISBN format (basic check: 10 or 13 digits)
    isbn = data['isbn']
    if not re.match(r'^\d{10}|\d{13}$', isbn):
        return None
    
    return data

def book_to_dict(book: Book) -> dict:
    return asdict(book)

def dict_to_book(data: dict) -> Book:
    return Book(
        id=data['id'],
        title=data['title'],
        author=data['author'],
        isbn=data['isbn'],
        price=data['price']
    )

@app.route('/books', methods=['GET'])
def get_books():
    page = request.args.get('page', 1, type=int)
    per_page = request.args.get('per_page', 10, type=int)
    
    # Validate pagination
    if page < 1 or per_page < 1:
        return jsonify({"error": "Invalid pagination parameters"}), 400
    
    start = (page - 1) * per_page
    end = start + per_page
    paginated_books = books_db[start:end]
    
    return jsonify(paginated_books), 200

@app.route('/books/<int:book_id>', methods=['GET'])
def get_book(book_id: int):
    book = next((b for b in books_db if b['id'] == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(book), 200

@app.route('/books', methods=['POST'])
def create_book():
    global next_id
    
    data = request.get_json()
    if not data:
        return jsonify({"error": "No JSON data provided"}), 400
    
    validated_data = validate_book_data(data)
    if not validated_data:
        return jsonify({"error": "Invalid book data"}), 422
    
    new_book = dict_to_book({
        **validated_data,
        'id': next_id
    })
    
    books_db.append(asdict(new_book))
    next_id += 1
    
    return jsonify(asdict(new_book)), 201

@app.route('/books/<int:book_id>', methods=['PUT'])
def update_book(book_id: int):
    book = next((b for b in books_db if b['id'] == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    data = request.get_json()
    if not data:
        return jsonify({"error": "No JSON data provided"}), 400
    
    # Merge update data with existing book data
    update_data = validate_book_data(data)
    if not update_data:
        return jsonify({"error": "Invalid update data"}), 422
    
    # Update fields that exist in the request
    for key in ['title', 'author', 'isbn', 'price']:
        if key in update_data:
            book[key] = update_data[key]
    
    return jsonify(book), 200

@app.route('/books/<int:book_id>', methods=['DELETE'])
def delete_book(book_id: int):
    global books_db
    
    book = next((b for b in books_db if b['id'] == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    books_db = [b for b in books_db if b['id'] != book_id]
    return jsonify({"message": "Book deleted successfully"}), 200

if __name__ == '__main__':
    app.run(debug=True)
