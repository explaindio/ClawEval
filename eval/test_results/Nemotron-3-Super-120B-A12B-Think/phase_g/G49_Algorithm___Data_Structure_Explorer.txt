import time
import collections
from typing import Any, Dict, List, Tuple, Optional

class LRUCache:
    def __init__(self, capacity: int):
        self._capacity: int = capacity
        self._cache: collections.OrderedDict[str, Tuple[Any, Optional[float]]] = collections.OrderedDict()
        self._hits: int = 0
        self._misses: int = 0
        self._evictions: int = 0
        self._expirations: int = 0

    def _is_expired(self, expiry: Optional[float]) -> bool:
        return expiry is not None and time.time() >= expiry

    def get(self, key: str) -> Any:
        if key in self._cache:
            value, expiry = self._cache[key]
            if self._is_expired(expiry):
                del self._cache[key]
                self._expirations += 1
                self._misses += 1
                return None
            self._cache.move_to_end(key)
            self._hits += 1
            return value
        self._misses += 1
        return None

    def put(self, key: str, value: Any, ttl: Optional[float] = None) -> None:
        expiry: Optional[float] = None if ttl is None else time.time() + ttl
        if key in self._cache:
            del self._cache[key]
        self._cache[key] = (value, expiry)
        self._cache.move_to_end(key)
        while len(self._cache) > self._capacity:
            old_key, (_, old_expiry) = self._cache.popitem(last=False)
            if self._is_expired(old_expiry):
                self._expirations += 1
            else:
                self._evictions += 1

    def delete(self, key: str) -> bool:
        if key in self._cache:
            del self._cache[key]
            return True
        return False

    def size(self) -> int:
        now = time.time()
        cnt = 0
        for _, expiry in self._cache.values():
            if expiry is None or now < expiry:
                cnt += 1
        return cnt

    def get_stats(self) -> Dict[str, int]:
        return {
            "hits": self._hits,
            "misses": self._misses,
            "evictions": self._evictions,
            "expirations": self._expirations,
        }

    def keys(self) -> List[str]:
        now = time.time()
        result: List[str] = []
        for key in reversed(self._cache):
            _, expiry = self._cache[key]
            if expiry is None or now < expiry:
                result.append(key)
        return result

    def clear(self) -> None:
        self._cache.clear()
        self._hits = self._misses = self._evictions = self._expirations = 0
