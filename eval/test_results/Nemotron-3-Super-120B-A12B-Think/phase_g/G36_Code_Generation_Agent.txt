import time
import threading
import collections

class RateLimiter:
    def __init__(self, max_requests: int, window_seconds: float, burst: int = 0):
        self.max_requests = max_requests
        self.window_seconds = max(window_seconds, 0.0)
        self.burst = max(burst, 0)
        self._data = {}  # (user_id, endpoint) -> state dict
        self._lock = threading.RLock()

    def _get_entry(self, key, now):
        entry = self._data.get(key)
        if entry is None:
            entry = {
                'timestamps': collections.deque(),
                'normal_tokens': float(self.max_requests),
                'burst_tokens': float(self.burst),
                'last_refill': now,
                'last_seen': now
            }
            self._data[key] = entry
        else:
            entry['last_seen'] = now
        return entry

    def _refill(self, entry, now):
        elapsed = now - entry['last_refill']
        if elapsed <= 0:
            return
        # normal token refill
        if self.window_seconds > 0:
            normal_rate = self.max_requests / self.window_seconds
            entry['normal_tokens'] = min(
                self.max_requests,
                entry['normal_tokens'] + normal_rate * elapsed
            )
        # burst token refill (2x window)
        if self.burst > 0 and self.window_seconds > 0:
            burst_rate = self.burst / (2 * self.window_seconds)
            entry['burst_tokens'] = min(
                self.burst,
                entry['burst_tokens'] + burst_rate * elapsed
            )
        entry['last_refill'] = now

    def allow(self, user_id: str, endpoint: str = "default") -> dict:
        now = time.time()
        key = (user_id, endpoint)
        with self._lock:
            entry = self._get_entry(key, now)
            self._refill(entry, now)

            # prune sliding window
            window_start = now - self.window_seconds
            dq = entry['timestamps']
            while dq and dq[0] < window_start:
                dq.popleft()
            used_normal = len(dq)

            if used_normal < self.max_requests:
                dq.append(now)
                remaining = self.max_requests - used_normal - 1
                return {"allowed": True, "remaining": remaining, "retry_after": None}
            else:
                # need burst
                if entry['burst_tokens'] >= 1:
                    entry['burst_tokens'] -= 1
                    remaining_burst = int(entry['burst_tokens'])
                    return {"allowed": True, "remaining": remaining_burst, "retry_after": None}
                else:
                    # compute retry_after
                        # time until oldest request leaves window
                    retry_normal = 0.0
                    if dq:
                        oldest = dq[0]
                        retry_normal = max(0.0, oldest + self.window_seconds - now)
                    # time until we get at least one burst token
                    retry_burst = 0.0
                    if entry['burst_tokens'] < 1:
                        needed = 1 - entry['burst_tokens']
                        if self.burst > 0 and self.window_seconds > 0:
                            burst_rate = self.burst / (2 * self.window_seconds)
                            retry_burst = needed / burst_rate if burst_rate > 0 else float('inf')
                        else:
                            retry_burst = float('inf')
                    retry_after = min(retry_normal, retry_burst)
                    return {"allowed": False, "remaining": 0, "retry_after": retry_after}

    def allow_priority(self, user_id: str, endpoint: str = "default") -> dict:
        # priority users bypass limits; state unchanged
        return {"allowed": True, "remaining": self.max_requests, "retry_after": None}

    def cleanup(self, max_age_seconds: float = None) -> int:
        if max_age_seconds is None:
            max_age_seconds = 2 * self.window_seconds
        now = time.time()
        removed = 0
        with self._lock:
            to_del = []
            for k, v in self._data.items():
                last_seen = v.get('last_seen', v.get('last_refill', 0))
                if now - last_seen > max_age_seconds:
                    to_del.append(k)
            for k in to_del:
                del self._data[k]
                removed += 1
        return removed

    def get_usage(self, user_id: str, endpoint: str = "default") -> dict:
        now = time.time()
        key = (user_id, endpoint)
        with self._lock:
            entry = self._data.get(key)
            if entry is None:
                used = 0
                burst_remaining = self.burst
            else:
                # prune timestamps for accurate usage
                window_start = now - self.window_seconds
                dq = entry['timestamps']
                while dq and dq[0] < window_start:
                    dq.popleft()
                used = len(dq)
                burst_remaining = int(entry['burst_tokens'])
            return {
                "used": used,
                "limit": self.max_requests,
                "window_seconds": self.window_seconds,
                "burst_remaining": burst_remaining
            }

    def to_dict(self) -> dict:
        with self._lock:
            data = {}
            for (uid, ep), entry in self._data.items():
                key_str = f"{uid}::{ep}"
                data[key_str] = {
                    'timestamps': list(entry['timestamps']),
                    'normal_tokens': entry['normal_tokens'],
                    'burst_tokens': entry['burst_tokens'],
                    'last_refill': entry['last_refill'],
                    'last_seen': entry['last_seen']
                }
            return {
                'max_requests': self.max_requests,
                'window_seconds': self.window_seconds,
                'burst': self.burst,
                'data': data
            }

    @classmethod
    def from_dict(cls, data: dict):
        obj = cls(data['max_requests'], data['window_seconds'], data.get('burst', 0))
        obj._data = {}
        for key_str, entry in data['data'].items():
            uid, ep = key_str.split('::', 1)
            key = (uid, ep)
            obj._data[key] = {
                'timestamps': collections.deque(entry['timestamps']),
                'normal_tokens': entry['normal_tokens'],
                'burst_tokens': entry['burst_tokens'],
                'last_refill': entry['last_refill'],
                'last_seen': entry['last_seen']
            }
        return obj
