import pytest
from your_module import calculate_tax  # Replace 'your_module' with actual module name

class TestCalculateTax:
    """Unit tests for calculate_tax function covering all edge cases."""

    def test_zero_income_single(self):
        """Test with zero income for single filer."""
        tax, rate = calculate_tax(0, 'single')
        assert tax == 0.0
        assert rate == 0

    def test_zero_income_married(self):
        """Test with zero income for married filer."""
        tax, rate = calculate_tax(0, 'married')
        assert tax == 0.0
        assert rate == 0

    def test_typical_single_income(self):
        """Test typical income within second bracket for single filer."""
        tax, rate = calculate_tax(30000, 'single')
        assert tax == 3000.0  # 11600*0.10 + (30000-11600)*0.12 = 1160 + 2208 = 3368
        assert tax == pytest.approx(3368.0, rel=0.01)
        assert rate == pytest.approx(11.23, rel=0.01)

    def test_typical_married_income(self):
        """Test typical income within second bracket for married filer."""
        tax, rate = calculate_tax(50000, 'married')
        # 23200*0.10 + (50000-23200)*0.12 = 2320 + 3216 = 5536
        assert tax == pytest.approx(5536.0, rel=0.01)
        assert rate == pytest.approx(11.07, rel=0.01)

    def test_high_income_single(self):
        """Test high income in top bracket for single filer."""
        tax, rate = calculate_tax(1000000, 'single')
        assert tax > 0
        assert rate == pytest.approx(37.0, rel=0.01)

    def test_high_income_married(self):
        """Test high income in top bracket for married filer."""
        tax, rate = calculate_tax(1000000, 'married')
        assert tax > 0

            calculate_tax(-1000, 'single')

        """Test with string income should raise ValueError."""
