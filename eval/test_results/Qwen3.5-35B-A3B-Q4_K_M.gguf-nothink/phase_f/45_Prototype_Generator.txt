```python
from flask import Flask, request, jsonify
from dataclasses import dataclass
from typing import List, Optional
import uuid

app = Flask(__name__)

@dataclass
class Book:
    id: str
    title: str
    author: str
    isbn: str
    price: float

    def to_dict(self):
        return {
            "id": self.id,
            "title": self.title,
            "author": self.author,
            "isbn": self.isbn,
            "price": self.price
        }

# In-memory database
books_db: List[Book] = []

def validate_book_data(data, required_fields):
    errors = []
    for field in required_fields:
        if field not in data or data[field] is None or data[field] == "":
            errors.append(f"{field} is required")
    if "price" in data:
        try:
            float(data["price"])
        except (ValueError, TypeError):
            errors.append("price must be a number")
    return errors

@app.route('/books', methods=['GET'])
def get_books():
    page = request.args.get('page', 1, type=int)
    per_page = request.args.get('per_page', 10, type=int)
    
    if page < 1:
        page = 1
    if per_page < 1:
        per_page = 10
        
    start = (page - 1) * per_page
    end = start + per_page
    
    paginated_books = books_db[start:end]
    
    return jsonify({
        "books": [book.to_dict() for book in paginated_books],
        "total": len(books_db),
        "page": page,
        "per_page": per_page
    }), 200

@app.route('/books/<book_id>', methods=['GET'])
def get_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(book.to_dict()), 200

@app.route('/books', methods=['POST'])
def create_book():
    data = request.get_json()
    if not data:
        return jsonify({"error": "Invalid JSON data"}), 400
    
    required_fields = ['title', 'author', 'isbn', 'price']
    validation_errors = validate_book_data(data, required_fields)
    
    if validation_errors:
        return jsonify({"error": validation_errors}), 400
    
    new_book = Book(
        id=str(uuid.uuid4()),
        title=data['title'],
        author=data['author'],
        isbn=data['isbn'],
        price=float(data['price'])
    )
    
    books_db.append(new_book)
    return jsonify(new_book.to_dict()), 201

@app.route('/books/<book_id>', methods=['PUT'])
def update_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    data = request.get_json()
    if not data:
        return jsonify({"error": "Invalid JSON data"}), 400
    
    required_fields = ['title', 'author', 'isbn', 'price']
    validation_errors = validate_book_data(data, required_fields)
    
    if validation_errors:
        return jsonify({"error": validation_errors}), 400
    
    book.title = data['title']
    book.author = data['author']
    book.isbn = data['isbn']
    book.price = float(data['price'])
    
    return jsonify(book.to_dict()), 200

@app.route('/books/<book_id>', methods=['DELETE'])
def delete_book(book_id):
    global books_db
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    books_db = [b for b in books_db if b.id != book_id]
    return jsonify({"message": "Book deleted successfully"}), 200

if __name__ == '__main__':
    app.run(debug=True)
```
