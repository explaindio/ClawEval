from flask import Flask, jsonify, request
from dataclasses import dataclass
import math

app = Flask(__name__)

# In-memory "database"
books_db = []
book_id_counter = 0

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

def next_id():
    global book_id_counter
    book_id_counter += 1
    return book_id_counter

def find_book(book_id):
    return next((b for b in books_db if b["id"] == book_id), None)

def validate_book(payload, partial=False):
    errors = []
    required = ["title", "author", "isbn", "price"]

    if partial:
        # At least one field must be provided
        if not any(k in payload for k in required):
            errors.append("At least one field must be provided")
        # Validate provided fields
        if "price" in payload:
            try:
                price = float(payload["price"])
                if price < 0:
                    errors.append("Price must be non‑negative")
            except (ValueError, TypeError):
                errors.append("Price must be a valid number")
        if "isbn" in payload and not isinstance(payload["isbn"], str):
            errors.append("ISBN must be a string")
        return errors

    # Full validation for POST (all fields required)
    for field in required:
        if field not in payload or payload[field] is None:
            errors.append(f"Missing required field: {field}")
    if "price" in payload:
        try:
            price = float(payload["price"])
            if price < 0:
                errors.append("Price must be non‑negative")
        except (ValueError, TypeError):
            errors.append("Price must be a valid number")
    if "isbn" in payload and not isinstance(payload["isbn"], str):
        errors.append("ISBN must be a string")
    return errors

@app.route("/books", methods=["GET"])
def list_books():
    page = request.args.get("page", 1, type=int)
    per_page = request.args.get("per_page", 10, type=int)

    if page < 1:
        page = 1
    if per_page < 1:
        per_page = 10

    total = len(books_db)
    start = (page - 1) * per_page
    end = start + per_page
    books = books_db[start:end]

    pages = math.ceil(total / per_page) if per_page > 0 else 0

    return jsonify({
        "books": books,
        "pagination": {
            "page": page,
            "per_page": per_page,
            "total": total,
            "pages": pages
        }
    }), 200

@app.route("/books/<int:book_id>", methods=["GET"])
def get_book(book_id):
    book = find_book(book_id)
    if book is None:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(book), 200

@app.route("/books", methods=["POST"])
def create_book():
    data = request.get_json(silent=True)
    if data is None:
        return jsonify({"error": "Request body must be JSON"}), 400

    errors = validate_book(data, partial=False)
    if errors:
        return jsonify({"error": errors[0]}), 422

    new_book = {
        "id": next_id(),
        "title": data["title"],
        "author": data["author"],
        "isbn": data["isbn"],
        "price": float(data["price"])
    }
    books_db.append(new_book)
    return jsonify(new_book), 201

@app.route("/books/<int:book_id>", methods=["PUT"])
def update_book(book_id):
    book = find_book(book_id)
    if book is None:
        return jsonify({"error": "Book not found"}), 404

    data = request.get_json(silent=True)
    if data is None:
        return jsonify({"error": "Request body must be JSON"}), 400

    errors = validate_book(data, partial=True)
    if errors:
        return jsonify({"error": errors[0]}), 422

    if "title" in data:
        book["title"] = data["title"]
    if "author" in data:
        book["author"] = data["author"]
    if "isbn" in data:
        book["isbn"] = data["isbn"]
    if "price" in data:
        price = float(data["price"])
        if price < 0:
            return jsonify({"error": "Price must be non‑negative"}), 422
        book["price"] = price

    return jsonify(book), 200

@app.route("/books/<int:book_id>", methods=["DELETE"])
def delete_book(book_id):
    book = find_book(book_id)
    if book is None:
        return jsonify({"error": "Book not found"}), 404
    books_db.remove(book)
    return jsonify({"message": "Book deleted successfully"}), 200

if __name__ == "__main__":
    app.run(debug=True)
