```python
from flask import Flask, request, jsonify
from dataclasses import dataclass
from typing import List

app = Flask(__name__)

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

books_db: List[Book] = []
next_id: int = 1

def make_error_response(message: str, status_code: int):
    return jsonify({"error": message}), status_code

@app.route('/books', methods=['GET'])
def get_books():
    page = request.args.get('page', 1, type=int)
    per_page = request.args.get('per_page', 10, type=int)
    
    if page < 1 or per_page < 1:
        return make_error_response("Invalid pagination parameters", 400)
    
    start = (page - 1) * per_page
    end = start + per_page
    paginated_books = books_db[start:end]
    
    return jsonify([book.__dict__ for book in paginated_books]), 200

@app.route('/books/<int:id>', methods=['GET'])
def get_book(id: int):
    book = next((b for b in books_db if b.id == id), None)
    if not book:
        return make_error_response("Book not found", 404)
    return jsonify(book.__dict__), 200

@app.route('/books', methods=['POST'])
def create_book():
    global next_id
    data = request.get_json()
    
    if not data:
        return make_error_response("Request body is required", 400)
    
    required_fields = ['title', 'author', 'isbn', 'price']
    for field in required_fields:
        if field not in data:
            return make_error_response(f"Missing required field: {field}", 422)
    
    try:
        price = float(data['price'])
    except (ValueError, TypeError):
        return make_error_response("Price must be a number", 422)
    
    book = Book(
        id=next_id,
        title=data['title'],
        author=data['author'],
        isbn=data['isbn'],
        price=price
    )
    next_id += 1
    books_db.append(book)
    
    return jsonify(book.__dict__), 201

@app.route('/books/<int:id>', methods=['PUT'])
def update_book(id: int):
    book = next((b for b in books_db if b.id == id), None)
    if not book:
        return make_error_response("Book not found", 404)
    
    data = request.get_json()
    if not data:
        return make_error_response("Request body is required", 400)
    
    if 'title' in data:
        book.title = data['title']
    if 'author' in data:
        book.author = data['author']
    if 'isbn' in data:
        book.isbn = data['isbn']
    if 'price' in data:
        try:
            book.price = float(data['price'])
        except (ValueError, TypeError):
            return make_error_response("Price must be a number", 422)
            
    return jsonify(book.__dict__), 200

@app.route('/books/<int:id>', methods=['DELETE'])
def delete_book(id: int):
    book = next((b for b in books_db if b.id == id), None)
    if not book:
        return make_error_response("Book not found", 404)
    
    books_db.remove(book)
    return jsonify({"message": "Book deleted"}), 200

if __name__ == '__main__':
    app.run(debug=True)
```
