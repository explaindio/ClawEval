```python
def parse_duration(s):
    s = s.strip()
    if not s:
        return -1
    
    total_seconds = 0
    current_num = 0
    has_digit = False
    
    for char in s:
        if char.isdigit():
            current_num = current_num * 10 + int(char)
            has_digit = True
        elif char in 'hms':
            if not has_digit:
                return -1
            if char == 'h':
                total_seconds += current_num * 3600
            elif char == 'm':
                total_seconds += current_num * 60
            elif char == 's':
                total_seconds += current_num
            current_num = 0
            has_digit = False
        else:
            return -1
            
    if has_digit:
        return -1
        
    return total_seconds
```
