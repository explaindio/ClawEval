```python
# test_calculate_tax.py
import pytest
from tax import calculate_tax  # adjust import as needed for your project

# ----------------------------------------------------------------------
# Zero income
# ----------------------------------------------------------------------
def test_zero_income_single():
    tax, rate = calculate_tax(0, 'single')
    assert tax == 0
    assert rate == 0

def test_zero_income_married():
    tax, rate = calculate_tax(0, 'married')
    assert tax == 0
    assert rate == 0

# ----------------------------------------------------------------------
# Typical incomes
# ----------------------------------------------------------------------
def test_typical_income_single():
    tax, rate = calculate_tax(30_000, 'single')
    assert tax == 3368.0
    assert rate == 11.23

def test_typical_income_married():
    tax, rate = calculate_tax(30_000, 'married')
    assert tax == 3136.0
    assert rate == 10.45

# ----------------------------------------------------------------------
# High incomes (above top bracket)
# ----------------------------------------------------------------------
def test_high_income_single():
    tax, rate = calculate_tax(1_000_000, 'single')
    assert tax == 328_187.75
    assert rate == 32.82

def test_high_income_married():
    tax, rate = calculate_tax(1_000_000, 'married')
    assert tax == 296_125.5
    assert rate == 29.61

# ----------------------------------------------------------------------
# Bracket boundary – single
# ----------------------------------------------------------------------
def test_bracket_boundary_single_at_limit():
    tax, rate = calculate_tax(11_600, 'single')
    assert tax == 1_160.0
    assert rate == 10.0

def test_bracket_boundary_single_just_above():
    tax, rate = calculate_tax(47_151, 'single')
    assert tax == 5_426.22
    assert rate == 11.51

def test_bracket_boundary_single_at_second_limit():
    tax, rate = calculate_tax(47_150, 'single')
    assert tax == 5_426.0
    assert rate == 11.51

# ----------------------------------------------------------------------
# Bracket boundary – married
# ----------------------------------------------------------------------
def test_bracket_boundary_married_at_limit():
    tax, rate = calculate_tax(23_200, 'married')
    assert tax == 2_320.0
    assert rate == 10.0

def test_bracket_boundary_married_just_above():
    tax, rate = calculate_tax(94_301, 'married')
    assert tax == 10_852.22
    assert rate == 11.51

def test_bracket_boundary_married_at_second_limit():
    tax, rate = calculate_tax(94_300, 'married')
    assert tax == 10_852.0
    assert rate == 11.51

# ----------------------------------------------------------------------
# Edge cases – floating‑point incomes, minimum taxable amount
# ----------------------------------------------------------------------
def test_float_income():
    tax, rate = calculate_tax(50_000.75, 'single')
    assert isinstance(tax, float)
    assert isinstance(rate, float)
    assert tax > 0

def test_minimum_taxable_income():
    tax, rate = calculate_tax(1, 'single')
    assert tax == 0.1
    assert rate == 10.0

# ----------------------------------------------------------------------
# Invalid inputs – filing status, negative income, non‑numeric income
# ----------------------------------------------------------------------
def test_invalid_filing_status():
    with pytest.raises(ValueError, match="Invalid filing status"):
        calculate_tax(50_000, 'head_of_household')

def test_negative_income():
    with pytest.raises(ValueError, match="Income must be non-negative"):
        calculate_tax(-1_000, 'single')

def test_non_numeric_income_string():
    with pytest.raises(ValueError, match="Income must be non-negative"):
        calculate_tax("50000", 'single')

def test_non_numeric_income_none():
    with pytest.raises(ValueError, match="Income must be non-negative"):
        calculate_tax(None, 'married')

# ----------------------------------------------------------------------
# Effective rate edge case – zero income yields 0%
# ----------------------------------------------------------------------
def test_effective_rate_zero_income():
    _, rate = calculate_tax(0, 'single')
    assert rate == 0
```
