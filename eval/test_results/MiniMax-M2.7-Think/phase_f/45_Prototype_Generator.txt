```python
from flask import Flask, jsonify, request
from dataclasses import dataclass
from typing import List, Optional

app = Flask(__name__)

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

# In‑memory "database"
books_db: List[dict] = []
next_id = 1

def validate_book_data(data: Optional[dict], partial: bool = False) -> Optional[str]:
    """Validate book payload.
    If partial is True only fields that are present are validated.
    Returns an error string or None if validation passes.
    """
    if not data:
        return "Request body must be a JSON object."
    required_fields = ["title", "author", "isbn", "price"]
    for field in required_fields:
        if field not in data:
            if partial:
                continue
            else:
                return f"Missing required field: {field}"
        if field == "price":
            try:
                float(data[field])
            except (TypeError, ValueError):
                return "Field 'price' must be a numeric value."
        else:
            if not isinstance(data[field], str) or not data[field].strip():
                return f"Field '{field}' must be a non‑empty string."
    return None

@app.route("/books", methods=["GET"])
def get_books():
    page = request.args.get("page", 1, type=int)
    per_page = request.args.get("per_page", 10, type=int)
    if page < 1:
        page = 1
    if per_page < 1:
        per_page = 1
    # Cap page size to avoid huge responses
    if per_page > 100:
        per_page = 100
    total = len(books_db)
    start = (page - 1) * per_page
    end = start + per_page
    page_books = books_db[start:end]
    return jsonify({
        "books": page_books,
        "total": total,
        "page": page,
        "per_page": per_page
    }), 200

@app.route("/books/<int:book_id>", methods=["GET"])
def get_book(book_id: int):
    book = next((b for b in books_db if b["id"] == book_id), None)
    if book is None:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(book), 200

@app.route("/books", methods=["POST"])
def create_book():
    if not request.is_json:
        return jsonify({"error": "Request must be JSON"}), 400
    data = request.get_json(silent=True)
    if data is None:
        return jsonify({"error": "Invalid JSON"}), 400
    err = validate_book_data(data, partial=False)
    if err:
        return jsonify({"error": err}), 422
    global next_id
    new_book = {
        "id": next_id,
        "title": data["title"].strip(),
        "author": data["author"].strip(),
        "isbn": data["isbn"].strip(),
        "price": float(data["price"])
    }
    books_db.append(new_book)
    next_id += 1
    return jsonify(new_book), 201

@app.route("/books/<int:book_id>", methods=["PUT"])
def update_book(book_id: int):
    book = next((b for b in books_db if b["id"] == book_id), None)
    if book is None:
        return jsonify({"error": "Book not found"}), 404
    if not request.is_json:
        return jsonify({"error": "Request must be JSON"}), 400
    data = request.get_json(silent=True)
    if data is None:
        return jsonify({"error": "Invalid JSON"}), 400
    err = validate_book_data(data, partial=True)
    if err:
        return jsonify({"error": err}), 422
    # Apply only provided fields
    if "title" in data:
        book["title"] = data["title"].strip()
    if "author" in data:
        book["author"] = data["author"].strip()
    if "isbn" in data:
        book["isbn"] = data["isbn"].strip()
    if "price" in data:
        book["price"] = float(data["price"])
    return jsonify(book), 200

@app.route("/books/<int:book_id>", methods=["DELETE"])
def delete_book(book_id: int):
    book = next((b for b in books_db if b["id"] == book_id), None)
    if book is None:
        return jsonify({"error": "Book not found"}), 404
    books_db.remove(book)
    return jsonify({"message": "Book deleted"}), 200

if __name__ == "__main__":
    app.run(debug=True)
```
