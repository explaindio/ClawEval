```python
from flask import Flask, request, jsonify
from dataclasses import dataclass, asdict, field
from typing import List

app = Flask(__name__)

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

# In-memory database
books_db: List[Book] = []
current_id_counter = 1

def validate_book_input(data):
    """Helper to validate book payload."""
    if not data:
        return {"error": "Request body must be JSON"}, 400
    
    required_fields = ['title', 'author', 'isbn', 'price']
    missing_fields = [f for f in required_fields if f not in data]
    
    if missing_fields:
        return {"error": f"Missing required fields: {', '.join(missing_fields)}"}, 422
    
    try:
        float(data['price'])
    except (ValueError, TypeError):
        return {"error": "Price must be a valid number"}, 422
    
    return None, None

@app.route('/books', methods=['GET'])
def get_books():
    page = request.args.get('page', 1, type=int)
    per_page = request.args.get('per_page', 10, type=int)
    
    start = (page - 1) * per_page
    end = start + per_page
    
    paginated_books = books_db[start:end]
    return jsonify([asdict(book) for book in paginated_books]), 200

@app.route('/books/<int:id>', methods=['GET'])
def get_book(id):
    book = next((b for b in books_db if b.id == id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(asdict(book)), 200

@app.route('/books', methods=['POST'])
def create_book():
    global current_id_counter
    data = request.get_json()
    
    error, status = validate_book_input(data)
    if error:
        return jsonify(error), status
    
    new_book = Book(
        id=current_id_counter,
        title=data['title'],
        author=data['author'],
        isbn=data['isbn'],
        price=float(data['price'])
    )
    
    books_db.append(new_book)
    current_id_counter += 1
    
    return jsonify(asdict(new_book)), 201

@app.route('/books/<int:id>', methods=['PUT'])
def update_book(id):
    data = request.get_json()
    error, status = validate_book_input(data)
    if error:
        return jsonify(error), status
    
    book = next((b for b in books_db if b.id == id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    book.title = data['title']
    book.author = data['author']
    book.isbn = data['isbn']
    book.price = float(data['price'])
    
    return jsonify(asdict(book)), 200

@app.route('/books/<int:id>', methods=['DELETE'])
def delete_book(id):
    global books_db
    book = next((b for b in books_db if b.id == id), None)
    
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    books_db = [b for b in books_db if b.id != id]
    return jsonify({"message": "Book deleted successfully"}), 200

if __name__ == '__main__':
    app.run(debug=True)
```
