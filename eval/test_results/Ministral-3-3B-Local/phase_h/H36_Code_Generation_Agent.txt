def merge_overlapping(intervals):
    if not intervals:
        return []

    intervals.sort(key=lambda x: (x[0], x[1]))
    merged = [intervals[0]]

    for current in intervals[1:]:
        last = merged[-1]
        if current[0] <= last[1]:
            last[1] = max(last[1], current[1])
        else:
            merged.append(current)

    return merged

def eval_rpn(tokens):
    stack = []
    for token in tokens:
        if token in '+-*/':
            b = stack.pop()
            a = stack.pop()
            if token == '+':
                stack.append(a + b)
            elif token == '-':
                stack.append(a - b)
            elif token == '*':
                stack.append(a * b)
            elif token == '/':
                stack.append(int(a / b))
        else:
            stack.append(int(token) if token.lstrip('-').isdigit() else int(token))
    return stack[0]

def deep_flatten(nested):
    flat = []
    def _flatten(item):
        if isinstance(item, (list, tuple)):
            for sub_item in item:
                _flatten(sub_item)
        else:
            flat.append(item)
    _flatten(nested)
    return flat
