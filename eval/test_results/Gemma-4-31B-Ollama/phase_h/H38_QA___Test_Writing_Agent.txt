def is_valid_password(password):
    if not (8 <= len(password) <= 64):
        return False
    if " " in password:
        return False
    if password[0].isdigit():
        return False
    if "password" in password.lower():
        return False
    
    has_upper = False
    has_lower = False
    has_digit = False
    has_special = False
    special_chars = "!@#$%^&*()-_+="
    
    for i in range(len(password)):
        char = password[i]
        if char.isupper(): has_upper = True
        if char.islower(): has_lower = True
        if char.isdigit(): has_digit = True
        if char in special_chars: has_special = True
        
        # Check for more than 3 consecutive identical characters
        if i >= 3:
            if char == password[i-1] == password[i-2] == password[i-3]:
                return False
                
    return all([has_upper, has_lower, has_digit, has_special])

def run_tests():
    tests = [
        ("Valid1@Pass", True),            # Valid
        ("Short1!", False),               # Too short
        ("A" * 65 + "1a!", False),        # Too long
        ("alllowercase1!", False),        # No uppercase
        ("ALLUPPERCASE1!", False),        # No lowercase
        ("NoDigitSpecial!", False),       # No digit
        ("NoSpecial123!", False),         # No special (if ! is not in list, but it is)
        ("NoSpecial123aB", False),        # No special
        ("Valid 1@Pass", False),          # Contains space
        ("1Valid@Pass", False),           # Starts with digit
        ("Valid1111@Pass", False),        # 4 consecutive identical chars
        ("MyPassword123!", False),        # Contains "password"
        ("mypassword123!", False),        # Contains "password" (case-insensitive)
        ("Valid1@P", False),              # Too short (7 chars)
        ("V@l1d_Pass_Long_Enough", True), # Valid
    ]
    
    results = []
    for pwd, expected in tests:
        actual = is_valid_password(pwd)
        results.append((pwd, expected, actual))
    return results
