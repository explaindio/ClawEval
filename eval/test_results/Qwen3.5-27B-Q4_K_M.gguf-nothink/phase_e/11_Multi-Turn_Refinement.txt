import re

def parse_duration(duration_str):
    if not duration_str:
        return -1
    
    pattern = r'(\d+)([hms])'
    matches = re.findall(pattern, duration_str)
    
    if not matches:
        return -1
    
    # Check if the entire string is consumed by the pattern
    reconstructed = ''.join(f"{num}{unit}" for num, unit in matches)
    if reconstructed != duration_str:
        return -1
    
    total_seconds = 0
    for num, unit in matches:
        value = int(num)
        if unit == 'h':
            total_seconds += value * 3600
        elif unit == 'm':
            total_seconds += value * 60
        elif unit == 's':
            total_seconds += value
    
    return total_seconds

```python
import re

def parse_duration(duration_str):
    if not duration_str:
        return -1
    
    pattern = r'(\d+)([hms])'
    matches = re.findall(pattern, duration_str)
    
    if not matches:
        return -1
    
    # Check if the entire string is consumed by the pattern
    reconstructed = ''.join(f"{num}{unit}" for num, unit in matches)
    if reconstructed != duration_str:
        return -1
    
    # Check for duplicate units
    units = [unit for _, unit in matches]
    if len(units) != len(set(units)):
        return -1
    
    total_seconds = 0
    for num, unit in matches:
        value = int(num)
        if unit == 'h':
            total_seconds += value * 3600
        elif unit == 'm':
            total_seconds += value * 60
        elif unit == 's':
            total_seconds += value
    
    return total_seconds
```
