from collections import OrderedDict

def lru_cache(capacity):
    class LRUCache:
        def __init__(self):
            self.cap = capacity
            self.store = OrderedDict()

        def get(self, key):
            if key in self.store:
                # Move to the end to mark as most recently used
                self.store.move_to_end(key)
                return self.store[key]
            return -1

        def put(self, key, value):
            # If key already exists, update value and move to end
            if key in self.store:
                self.store[key] = value
                self.store.move_to_end(key)
                return

            # If at capacity, evict the least recently used item
            if self.cap > 0 and len(self.store) >= self.cap:
                self.store.popitem(last=False)  # Remove the first (oldest) item

            # Add new item only if capacity is positive
            if self.cap > 0:
                self.store[key] = value

    return LRUCache()
