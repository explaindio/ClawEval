def is_valid_password(password):
    # Length check
    if len(password) < 8 or len(password) > 64:
        return False
    # No spaces
    if ' ' in password:
        return False
    # Cannot start with a digit
    if password and password[0].isdigit():
        return False
    # At least one uppercase letter
    if not any(c.isupper() for c in password):
        return False
    # At least one lowercase letter
    if not any(c.islower() for c in password):
        return False
    # At least one digit
    if not any(c.isdigit() for c in password):
        return False
    # At least one special character
    specials = "!@#$%^&*()-_+="
    if not any(c in specials for c in password):
        return False
    # No more than 3 consecutive identical characters
    prev = None
    count = 0
    for c in password:
        if c == prev:
            count += 1
        else:
            prev = c
            count = 1
        if count > 3:
            return False
    # Cannot contain the word "password" (case‑insensitive)
    if "password" in password.lower():
        return False
    return True


def run_tests():
    test_cases = [
        ("Abcdef1!@#", True),
        ("A1!b", False),
        ("A" * 65 + "b", False),
        ("abcdef1!@#", False),
        ("ABCDEF1!@#", False),
        ("Abcdef!@#", False),
        ("Abcdef123", False),
        ("Abc def1!@#", False),
        ("1Abcdef!@#", False),
        ("Abcdeeee1!@#", False),
        ("ThisIsPassword123!", False),
        ("Abcdeee1!@#", True),
        ("!Abcdef12", True),
        ("AAAAbbb1!@#", False),
        ("Abcde f1!@#", False),
    ]
    results = []
    for inp, expected in test_cases:
        actual = is_valid_password(inp)
        results.append((inp, expected, actual))
    return results
