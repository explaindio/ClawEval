```python
from flask import Flask, request, jsonify
from dataclasses import dataclass, asdict
from typing import List
import uuid

app = Flask(__name__)

@dataclass
class Book:
    id: str
    title: str
    author: str
    isbn: str
    price: float

books_db: List[Book] = []

def error_response(message: str, status_code: int):
    return jsonify({"error": message}), status_code

@app.route('/books', methods=['GET'])
def get_books():
    try:
        page = request.args.get('page', 1, type=int)
        per_page = request.args.get('per_page', 10, type=int)
    except ValueError:
        return error_response("Invalid pagination parameters", 400)

    if page < 1 or per_page < 1:
        return error_response("Invalid pagination parameters", 400)

    start = (page - 1) * per_page
    end = start + per_page
    paginated_books = books_db[start:end]

    return jsonify({
        "books": [asdict(book) for book in paginated_books],
        "total": len(books_db),
        "page": page,
        "per_page": per_page
    }), 200

@app.route('/books/<book_id>', methods=['GET'])
def get_book(book_id: str):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return error_response("Book not found", 404)
    return jsonify(asdict(book)), 200

@app.route('/books', methods=['POST'])
def create_book():
    data = request.get_json()
    if not data:
        return error_response("Request body must be JSON", 422)

    required_fields = ['title', 'author', 'isbn', 'price']
    missing_fields = [f for f in required_fields if f not in data]
    if missing_fields:
        return error_response(f"Missing required fields: {', '.join(missing_fields)}", 400)

    try:
        price = float(data['price'])
    except (ValueError, TypeError):
        return error_response("Price must be a number", 422)

    new_book = Book(
        id=str(uuid.uuid4()),
        title=str(data['title']),
        author=str(data['author']),
        isbn=str(data['isbn']),
        price=price
    )
    books_db.append(new_book)
    return jsonify(asdict(new_book)), 201

@app.route('/books/<book_id>', methods=['PUT'])
def update_book(book_id: str):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return error_response("Book not found", 404)

    data = request.get_json()
    if not data:
        return error_response("Request body must be JSON", 422)

    if 'price' in data:
        try:
            data['price'] = float(data['price'])
        except (ValueError, TypeError):
            return error_response("Price must be a number", 422)

    if 'title' in data: book.title = str(data['title'])
    if 'author' in data: book.author = str(data['author'])
    if 'isbn' in data: book.isbn = str(data['isbn'])
    if 'price' in data: book.price = data['price']

    return jsonify(asdict(book)), 200

@app.route('/books/<book_id>', methods=['DELETE'])
def delete_book(book_id: str):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return error_response("Book not found", 404)

    books_db[:] = [b for b in books_db if b.id != book_id]
    return jsonify({"message": "Book deleted successfully"}), 200

if __name__ == '__main__':
    app.run(debug=True)
```
