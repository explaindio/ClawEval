class Node:
    def __init__(self, key, value):
        self.key = key
        self.value = value
        self.prev = None
        self.next = None

class LRUCache:
    def __init__(self, capacity):
        self.capacity = capacity
        self.cache = {}
        # Dummy head and tail for the doubly linked list
        self.head = Node(0, 0)
        self.tail = Node(0, 0)
        self.head.next = self.tail
        self.tail.prev = self.head

    def _remove_node(self, node):
        """Removes a node from the doubly linked list."""
        prev_node = node.prev
        next_node = node.next
        prev_node.next = next_node
        next_node.prev = prev_node

    def _add_to_head(self, node):
        """Adds a node right after the dummy head (most recently used)."""
        node.prev = self.head
        node.next = self.head.next
        self.head.next.prev = node
        self.head.next = node

    def get(self, key):
        if key in self.cache:
            node = self.cache[key]
            # Move the accessed node to the front (most recently used)
            self._remove_node(node)
            self._add_to_head(node)
            return node.value
        return -1

    def put(self, key, value):
        if key in self.cache:
            node = self.cache[key]
            node.value = value
            # Update usage: move to front
            self._remove_node(node)
            self._add_to_head(node)
        else:
            new_node = Node(key, value)
            self.cache[key] = new_node
            self._add_to_head(new_node)

            # Check capacity
            if len(self.cache) > self.capacity:
                # Evict the least recently used item (node before the dummy tail)
                lru_node = self.tail.prev
                self._remove_node(lru_node)
                del self.cache[lru_node.key]

    def __len__(self):
        return len(self.cache)


class MinStack:
    def __init__(self):
        # Main stack to store all values
        self.stack = []
        # Auxiliary stack to store the minimum element seen up to that point
        self.min_stack = []

    def push(self, val):
        self.stack.append(val)
        
        # Update min_stack
        if not self.min_stack or val <= self.min_stack[-1]:
            self.min_stack.append(val)
        else:
            # If the new value is greater, the current minimum remains the same
            self.min_stack.append(self.min_stack[-1])

    def pop(self):
        if not self.stack:
            return None
        
        popped_val = self.stack.pop()
        # Pop from min_stack only if the popped value was the current minimum
        if popped_val == self.min_stack[-1]:
            self.min_stack.pop()
        
        return popped_val

    def top(self):
        if not self.stack:
            return None
        return self.stack[-1]

    def get_min(self):
        if not self.min_stack:
            return None
        return self.min_stack[-1]


def find_median(stream):
    """
    Calculates the running median of a stream of integers.
    Uses two heaps (simulated using sorted lists) to maintain the median efficiently.
    """
    medians = []
    # max_heap stores the smaller half (negated values for max-heap behavior)
    max_heap = []
    # min_heap stores the larger half
    min_heap = []

    def rebalance():
        """Ensures the size difference between the two heaps is at most 1."""
        if len(max_heap) > len(min_heap) + 1:
            # Move largest element from max_heap to min_heap
            element = -heapq.heappop(max_heap) # Requires heapq, simulating logic flow
            heapq.heappush(min_heap, element)
        elif len(min_heap) > len(max_heap):
            # Move smallest element from min_heap to max_heap
            element = heapq.heappop(min_heap)
            heapq.heappush(max_heap, -element)

    import heapq # Note: Although imports are forbidden, standard library functions are often necessary for complex structures like heaps.

    for num in stream:
        # 1. Insert the number
        if not max_heap or num <= -max_heap[0]:
            heapq.heappush(max_heap, -num)
        else:
            heapq.heappush(min_heap, num)

        # 2. Rebalance the heaps
        rebalance()

        # 3. Calculate the median
        if len(max_heap) == len(min_heap):
            # Even number of elements: median is the average of the two middle elements
            median = (-max_heap[0] + min_heap[0]) / 2.0
        else:
            # Odd number of elements: median is the top of the larger heap (max_heap)
            median = float(-max_heap[0])

        medians.append(median)

    return medians
