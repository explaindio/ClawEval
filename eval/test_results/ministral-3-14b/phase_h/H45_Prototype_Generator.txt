        errors = []

        return {"valid": len(errors) == 0, "errors": errors}

    def _validate_data(self, data, schema, errors):
        if not isinstance(data, dict):
            return

        for field_name, field_schema in schema.items():
            if field_name not in data:
                if field_schema.get("required", False):
                    errors.append(f"Field '{field_name}' is required but missing")
                continue

            self._validate_field(field_value, field_schema, errors, field_name)

                errors.append(f"Field '{field_name}' must be a list")

### Explanation:
1. **Main Validation Method (`validate`)**:
   - Initializes an empty list to collect errors.
   - Calls the helper method `_validate_data` to start the validation process.
   - Returns a dictionary with `valid` (boolean) and `errors` (list of error strings).

2. **Helper Methods**:
   - `_validate_data`: Checks each field in the schema against the provided data.
   - `_validate_field`: Validates the type of each field and delegates further validation based on the type.
   - `_validate_string`, `_validate_number`, `_validate_list`, `_validate_dict`: Handle type-specific validations (e.g., length checks for strings/lists, min/max for numbers).

3. **Type Handling**:
   - **Strings**: Checks `min_length` and `max_length`.
   - **Numbers (int/float)**: Checks `min` and `max` values.
   - **Lists**: Checks `min_length` and `max_length` for the list itself (not the items).
   - **Dictionaries**: Currently does not validate nested fields (would require recursive validation).

4. **Error Handling**:
   - Collects all errors and returns them in the response.
   - Extra fields in the data are ignored (no error).

### Example Usage:
```python
validator = JSONValidator()
schema = {
    "name": {"type": "str", "required": True, "min_length": 3, "max_length": 20},
    "age": {"type": "int", "required": True, "min": 0, "max": 120},
    "is_active": {"type": "bool", "required": False},
    "tags": {"type": "list", "required": False, "min_length": 0, "max_length": 5}
}

data = {
    "name": "Alice",
    "age": 30,
    "is_active": True,
    "tags": ["python", "coding"]
}

print(validator.validate(data, schema))
# Output: {'valid': True, 'errors': []}

data_invalid = {
    "name": "A",  # Too short
    "age": 150,   # Too high
    "tags": ["python", "coding", "extra"]  # Too many
}

print(validator.validate(data_invalid, schema))
# Output: {'valid': False, 'errors': [
#     "Field 'name' must have at least 3 characters",
#     "Field 'age' must be at most 120",
#     "Field 'tags' must have at most 5 items"
# ]}
```
