from flask import Flask, request, jsonify
from dataclasses import dataclass, asdict
from typing import List, Optional
import re

app = Flask(__name__)

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

# In-memory database
books_db: List[Book] = []
next_id = 1

def validate_book_data(data: dict) -> Optional[dict]:
    """Validates required fields and basic formats."""
    required_fields = ['title', 'author', 'isbn', 'price']
    
    for field in required_fields:
        if field not in data:
            return {"error": f"Missing required field: {field}"}
    
    # Validate ISBN format (basic check: 10 or 13 digits)
    isbn = data['isbn']
    if not re.match(r'^\d{10}|\d{13}$', isbn):
        return {"error": "Invalid ISBN format. Must be 10 or 13 digits."}
    
    # Validate price is a positive number
    try:
        price = float(data['price'])
        if price <= 0:
            return {"error": "Price must be a positive number."}
        data['price'] = price
    except ValueError:
        return {"error": "Price must be a valid number."}
    
    return data

def get_book_by_id(book_id: int) -> Optional[Book]:
    for book in books_db:
        if book.id == book_id:
            return book
    return None

@app.route('/books', methods=['GET'])
def list_books():
    global next_id
    page = request.args.get('page', 1, type=int)
    per_page = request.args.get('per_page', 10, type=int)
    
    # Limit per_page to prevent abuse
    if per_page > 100:
        per_page = 100
    
    start_index = (page - 1) * per_page
    end_index = start_index + per_page
    
    paginated_books = books_db[start_index:end_index]
    
    return jsonify({
        "books": [asdict(book) for book in paginated_books],
        "page": page,
        "per_page": per_page,
        "total": len(books_db)
    }), 200

@app.route('/books/<int:book_id>', methods=['GET'])
def get_book(book_id: int):
    book = get_book_by_id(book_id)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(asdict(book)), 200

@app.route('/books', methods=['POST'])
def create_book():
    global next_id
    data = request.get_json()
    
    if not data:
        return jsonify({"error": "Request body must be JSON"}), 400
    
    validated_data = validate_book_data(data)
    if validated_data is None:
        return jsonify(validated_data), 422
    
    new_book = Book(
        id=next_id,
        title=validated_data['title'],
        author=validated_data['author'],
        isbn=validated_data['isbn'],
        price=validated_data['price']
    )
    
    books_db.append(new_book)
    next_id += 1
    
    return jsonify(asdict(new_book)), 201

@app.route('/books/<int:book_id>', methods=['PUT'])
def update_book(book_id: int):
    book = get_book_by_id(book_id)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    data = request.get_json()
    if not data:
        return jsonify({"error": "Request body must be JSON"}), 400
    
    # Merge provided data with existing book data
    update_data = validate_book_data(data)
    if update_data is None:
        return jsonify(update_data), 422
    
    # Update fields if provided
    if 'title' in data:
        book.title = data['title']
    if 'author' in data:
        book.author = data['author']
    if 'isbn' in data:
        book.isbn = data['isbn']
    if 'price' in data:
        book.price = data['price']
    
    return jsonify(asdict(book)), 200

@app.route('/books/<int:book_id>', methods=['DELETE'])
def delete_book(book_id: int):
    global books_db
    book = get_book_by_id(book_id)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    books_db = [b for b in books_db if b.id != book_id]
    return jsonify({"message": "Book deleted successfully"}), 200

if __name__ == '__main__':
    app.run(debug=True)
