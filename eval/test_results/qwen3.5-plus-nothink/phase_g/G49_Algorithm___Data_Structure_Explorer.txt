```python
import time
from collections import OrderedDict

class LRUCache:
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.cache = OrderedDict()  # key -> (value, expiry_time)
        self.hits = 0
        self.misses = 0
        self.evictions = 0
        self.expirations = 0

    def _is_expired(self, key: str) -> bool:
        if key not in self.cache:
            return True
        _, expiry = self.cache[key]
        if expiry is not None and time.time() > expiry:
            del self.cache[key]
            self.expirations += 1
            return True
        return False

    def get(self, key: str) -> any:
        if key not in self.cache:
            self.misses += 1
            return None
        
        if self._is_expired(key):
            self.misses += 1
            return None
        
        self.cache.move_to_end(key)
        self.hits += 1
        return self.cache[key][0]

    def put(self, key: str, value: any, ttl: float = None) -> None:
        expiry = None
        if ttl is not None:
            expiry = time.time() + ttl
        
        if key in self.cache:
            del self.cache[key]
        elif len(self.cache) >= self.capacity:
            oldest_key = next(iter(self.cache))
            del self.cache[oldest_key]
            self.evictions += 1
        
        self.cache[key] = (value, expiry)

    def delete(self, key: str) -> bool:
        if key in self.cache:
            if self._is_expired(key):
                return False
            del self.cache[key]
            return True
        return False

    def size(self) -> int:
        current_time = time.time()
        count = 0
        keys_to_remove = []
        
        for key, (value, expiry) in self.cache.items():
            if expiry is not None and current_time > expiry:
                keys_to_remove.append(key)
            else:
                count += 1
        
        for key in keys_to_remove:
            del self.cache[key]
            self.expirations += 1
            
        return count

    def get_stats(self) -> dict:
        return {
            "hits": self.hits,
            "misses": self.misses,
            "evictions": self.evictions,
            "expirations": self.expirations
        }

    def keys(self) -> list:
        current_time = time.time()
        valid_keys = []
        keys_to_remove = []
        
        for key, (value, expiry) in self.cache.items():
            if expiry is not None and current_time > expiry:
                keys_to_remove.append(key)
            else:
                valid_keys.append(key)
        
        for key in keys_to_remove:
            del self.cache[key]
            self.expirations += 1
            
        return valid_keys

    def clear(self) -> None:
        self.cache.clear()
        self.hits = 0
        self.misses = 0
        self.evictions = 0
        self.expirations = 0
```
