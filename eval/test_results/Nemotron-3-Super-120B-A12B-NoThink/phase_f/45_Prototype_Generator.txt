```python
from flask import Flask, request, jsonify
from dataclasses import dataclass, asdict
from typing import List, Optional

app = Flask(__name__)

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

books: List[Book] = []
_next_id = 1

def _find_book(book_id: int) -> Optional[Book]:
    return next((b for b in books if b.id == book_id), None)

def _validate_book_data(data: dict, partial: bool = False) -> tuple[bool, Optional[str]]:
    required = ["title", "author", "isbn", "price"]
    if not partial:
        missing = [f for f in required if f not in data]
        if missing:
            return False, f"Missing fields: {', '.join(missing)}"
    if "title" in data and not isinstance(data["title"], str):
        return False, "title must be a string"
    if "author" in data and not isinstance(data["author"], str):
        return False, "author must be a string"
    if "isbn" in data and not isinstance(data["isbn"], str):
        return False, "isbn must be a string"
    if "price" in data:
        try:
            price = float(data["price"])
            if price <= 0:
                return False, "price must be greater than zero"
        except (ValueError, TypeError):
            return False, "price must be a number"
    return True, None

@app.route("/books", methods=["GET"])
def list_books():
    try:
        page = int(request.args.get("page", 1))
        per_page = int(request.args.get("per_page", 10))
    except ValueError:
        return jsonify(error="page and per_page must be integers"), 400
    if page < 1 or per_page < 1:
        return jsonify(error="page and per_page must be positive"), 400
    start = (page - 1) * per_page
    end = start + per_page
    paginated = books[start:end]
    return jsonify({
        "page": page,
        "per_page": per_page,
        "total": len(books),
        "books": [asdict(b) for b in paginated]
    }), 200

@app.route("/books/<int:book_id>", methods=["GET"])
def get_book(book_id: int):
    book = _find_book(book_id)
    if book is None:
        return jsonify(error="Book not found"), 404
    return jsonify(asdict(book)), 200

@app.route("/books", methods=["POST"])
def create_book():
    global _next_id
    if not request.is_json:
        return jsonify(error="Request must be JSON"), 400
    data = request.get_json()
    valid, err = _validate_book_data(data)
    if not valid:
        return jsonify(error=err), 422
    book = Book(
        id=_next_id,
        title=data["title"],
        author=data["author"],
        isbn=data["isbn"],
        price=float(data["price"])
    )
    books.append(book)
    _next_id += 1
    return jsonify(asdict(book)), 201

@app.route("/books/<int:book_id>", methods=["PUT"])
def update_book(book_id: int):
    if not request.is_json:
        return jsonify(error="Request must be JSON"), 400
    book = _find_book(book_id)
    if book is None:
        return jsonify(error="Book not found"), 404
    data = request.get_json()
    valid, err = _validate_book_data(data, partial=True)
    if not valid:
        return jsonify(error=err), 422
    for field in ("title", "author", "isbn", "price"):
        if field in data:
            setattr(book, field, data[field] if field != "price" else float(data[field]))
    return jsonify(asdict(book)), 200

@app.route("/books/<int:book_id>", methods=["DELETE"])
def delete_book(book_id: int):
    global books
    book = _find_book(book_id)
    if book is None:
        return jsonify(error="Book not found"), 404
    books = [b for b in books if b.id != book_id]
    return "", 204

if __name__ == "__main__":
    app.run(debug=True)
```
