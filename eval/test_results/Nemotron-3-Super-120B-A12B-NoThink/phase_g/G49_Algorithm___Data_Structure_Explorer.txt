```python
import time
from collections import OrderedDict

class LRUCache:
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.cache = OrderedDict()
        self.hits = self.misses = self.evictions = self.expirations = 0

    def _is_expired(self, expiry):
        return expiry is not None and time.time() > expiry

    def get(self, key: str):
        if key in self.cache:
            value, expiry = self.cache[key]
            if self._is_expired(expiry):
                del self.cache[key]
                self.expirations += 1
                self.misses += 1
                return None
            self.cache.move_to_end(key)
            self.hits += 1
            return value
        self.misses += 1
        return None

    def put(self, key: str, value: any, ttl: float = None):
        if key in self.cache:
            del self.cache[key]
        expiry = None if ttl is None else time.time() + ttl
        self.cache[key] = (value, expiry)
        self.cache.move_to_end(key)
        if len(self.cache) > self.capacity:
            lru_key, (lru_value, lru_expiry) = self.cache.popitem(last=False)
            if self._is_expired(lru_expiry):
                self.expirations += 1
            else:
                self.evictions += 1

    def delete(self, key: str) -> bool:
        if key in self.cache:
            del self.cache[key]
            return True
        return False

    def size(self) -> int:
        now = time.time()
        count = 0
        for _, (_, expiry) in self.cache.items():
            if expiry is None or now <= expiry:
                count += 1
        return count

    def get_stats(self) -> dict:
        return {
            "hits": self.hits,
            "misses": self.misses,
            "evictions": self.evictions,
            "expirations": self.expirations
        }

    def keys(self) -> list:
        now = time.time()
        result = []
        for key in reversed(self.cache):
            _, expiry = self.cache[key]
            if expiry is None or now <= expiry:
                result.append(key)
        return result

    def clear(self) -> None:
        self.cache.clear()
        self.hits = self.misses = self.evictions = self.expirations = 0
```
