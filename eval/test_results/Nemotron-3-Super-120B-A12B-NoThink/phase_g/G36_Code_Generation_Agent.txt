import time
import threading
from collections import deque
from typing import Dict, Tuple, Any, List, Optional


class RateLimiter:
    def __init__(self, max_requests: int, window_seconds: float, burst: int = 0):
        self.max_requests: int = max_requests
        self.window_seconds: float = window_seconds
        self.burst: int = burst
        self._data: Dict[Tuple[str, str], Dict[str, Any]] = {}
        self._lock = threading.RLock()

    def _init_entry(self, now: float) -> Dict[str, Any]:
        return {
            "timestamps": deque(),
            "burst_tokens": float(self.burst),
            "burst_last_refill": now,
            "last_seen": now,
        }

    def _prune_timestamps(self, timestamps: deque, now: float) -> None:
        window = self.window_seconds
        while timestamps and now - timestamps[0] > window:
            timestamps.popleft()

    def _refill_burst(self, entry: Dict[str, Any], now: float) -> None:
        if self.burst == 0:
            return
        elapsed = now - entry["burst_last_refill"]
        if elapsed <= 0:
            return
        rate = self.burst / (2 * self.window_seconds)
        if rate <= 0:
            return
        to_add = elapsed * rate
        entry["burst_tokens"] = min(self.burst, entry["burst_tokens"] + to_add)
        entry["burst_last_refill"] = now

    def allow(self, user_id: str, endpoint: str = "default") -> dict:
        now = time.time()
        key = (user_id, endpoint)
        with self._lock:
            entry = self._data.get(key)
            if entry is None:
                entry = self._init_entry(now)
                self._data[key] = entry
            else:
                self._refill_burst(entry, now)
                self._prune_timestamps(entry["timestamps"], now)
                entry["last_seen"] = now

            used_normal = len(entry["timestamps"])
            if used_normal < self.max_requests:
                entry["timestamps"].append(now)
                remaining = self.max_requests - used_normal - 1
                return {"allowed": True, "remaining": remaining, "retry_after": None}
            else:
                if entry["burst_tokens"] >= 1.0:
                    entry["burst_tokens"] -= 1.0
                    remaining = int(entry["burst_tokens"])
                    return {"allowed": True, "remaining": remaining, "retry_after": None}
                else:
                    # compute retry_after
                    timestamps = entry["timestamps"]
                    if timestamps:
                        oldest = timestamps[0]
                        time_to_normal = max(0.0, oldest + self.window_seconds - now)
                    else:
                        time_to_normal = 0.0

                    if self.burst > 0:
                        rate_burst = self.burst / (2 * self.window_seconds)
                        if rate_burst > 0:
                            needed = 1.0 - entry["burst_tokens"]
                            time_to_burst = needed / rate_burst if needed > 0 else 0.0
                        else:
                            time_to_burst = float('inf')
                    else:
                        time_to_burst = float('inf')

                    retry_after = min(time_to_normal, time_to_burst)
                    if retry_after == float('inf'):
                        retry_after = None
                    else:
                        retry_after = max(0.0, retry_after)
                    return {"allowed": False, "remaining": 0, "retry_after": retry_after}

    def allow_priority(self, user_id: str, endpoint: str = "default") -> dict:
        # Priority users bypass limits; state unchanged.
        return {"allowed": True, "remaining": self.max_requests, "retry_after": None}

    def cleanup(self, max_age_seconds: float = None) -> int:
        if max_age_seconds is None:
            max_age_seconds = 2 * self.window_seconds
        now = time.time()
        removed = 0
        with self._lock:
            to_delete = [
                key
                for key, entry in self._data.items()
                if now - entry["last_seen"] > max_age_seconds
            ]
            for key in to_delete:
                del self._data[key]
                removed += 1
        return removed

    def get_usage(self, user_id: str, endpoint: str = "default") -> dict:

        return {

                "data": data_serialized,
