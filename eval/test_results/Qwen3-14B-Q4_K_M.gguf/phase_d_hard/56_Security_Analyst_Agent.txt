Security audit this code. There are 7 vulnerabilities. Most are obvious, but ONE is subtle (prototype pollution). Find all 7.

```javascript
const express = require('express');
const app = express();
app.use(express.json());

// Utility: deep merge objects
function merge(target, source) {
  for (let key in source) {
    if (typeof source[key] === 'object' && source[key] !== null) {
      target[key] = target[key] || {};
      merge(target[key], source[key]);
    } else {
      target[key] = source[key];
    }
  }
  return target;
}

const users = {};

app.post('/register', (req, res) => {
  const defaults = { role: 'user', active: true };
  const userData = merge(defaults, req.body);
  users[userData.email] = userData;
  res.json({ message: `Welcome ${userData.name}!`, role: userData.role });
});

app.get('/user/:email', (req, res) => {
  const user = users[req.params.email];
  if (!user) return res.status(404).send('Not found');
  res.json(user);
});

app.post('/admin/config', (req, res) => {
  if (users[req.body.email]?.role !== 'admin') {
    return res.status(403).send('Forbidden');
  }
  const config = require(req.body.configPath);
  res.json(config);
});

app.listen(3000);
```

## RESPONSE:
### **Vulnerability 1: Prototype Pollution via `__proto__` (Subtle)**  
**OWASP Category:** Injection (A03:2021)  
**Severity:** High  
**Exploitation Example:**  
```http
POST /register
Content-Type: application/json

{
  "__proto__": {
    "role": "admin"
  }
}
```  
**Fix:**  
Use a deep clone of the `defaults` object before merging. Replace the `merge` function with a deep cloning approach to avoid modifying the original `defaults` object. Example:  
```javascript
function merge(target, source) {
  const clonedTarget = JSON.parse(JSON.stringify(target));
  for (let key in source) {
    if (typeof source[key] === 'object' && source[key] !== null) {
      clonedTarget[key] = clonedTarget[key] || {};
      merge(clonedTarget[key], source[key]);
    } else {
      clonedTarget[key] = source[key];
    }
  }
  return clonedTarget;
}
```

### **Subtle Vulnerability: Prototype Pollution via `__proto__`**  
**Exploitation Example:**  
```http
POST /register
Content-Type: application/json

{
  "__proto__": {
    "role": "admin"
  }
}
```  
**Explanation:**  
The `merge` function modifies the `defaults` object in place. If an attacker sends a payload with `__proto__`, it will overwrite the prototype of the `defaults` object. When the user is created, even if they don't specify a `role`, the prototype's `role` will be used, making the user an admin.  
**Fix:**  
Use a deep clone of the `defaults` object to prevent prototype pollution, as described in Vulnerability 1.
