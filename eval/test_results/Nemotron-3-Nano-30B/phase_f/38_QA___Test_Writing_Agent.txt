import pytestfrom calculate_tax import calculate_tax

def test_zero_income():
    assert calculate_tax(0, 'single') == (0.0, 0.0)

def test_negative_income_raises():
    with pytest.raises(ValueError):
        calculate_tax(-5, 'single')

def test_invalid_filing_status_raises():
    with pytest.raises(ValueError):
        calculate_tax(50000, 'head_of_household')

def test_non_numeric_income_raises():
    with pytest.raises(ValueError):
        calculate_tax('10000', 'single')

def test_single_bracket_boundary_11600():
    assert calculate_tax(11600, 'single') == (1160.0, 10.0)

def test_single_bracket_boundary_47150():
    tax, eff = calculate_tax(47150, 'single')
    assert round(tax, 2) == 5426.0
    assert round(eff, 2) == 11.51
